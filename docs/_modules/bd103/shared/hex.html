[docs]    def make_safe(string: str) -> str:
        if string.startswith("#"):
            string = string[1:]

        for i in string:
            if i not in Hex.key:
                raise Exception("Invalid Hex Code")

        return string.lower()



[docs]    def convert_char(char: str) -> int:
        return Hex.key.get(char)



[docs]    def to_decimal(self) -> int:
        sum = Hex.convert_char(self.og[0])

        if len(self.og) == 1:
            return sum

        for i in self.og[1:]:
            sum *= 16
            sum += Hex.convert_char(i)

        return sum
